### .bin 点云文件读取代码

KITTI 点云格式：每个点 4 个 float32 值（x, y, z, intensity）

```python
import numpy as np
import open3d as o3d

def load_kitti_bin(bin_path):
    """加载 KITTI .bin 格式点云"""
    points = np.fromfile(bin_path, dtype=np.float32).reshape(-1, 4)
    xyz = points[:, :3]          # 坐标 (N, 3)
    intensity = points[:, 3]     # 反射强度 (N,)
    return xyz, intensity

# 加载点云
xyz, intensity = load_kitti_bin("velodyne/000000.bin")
print(f"点数: {len(xyz)}, x范围: [{xyz[:,0].min():.1f}, {xyz[:,0].max():.1f}]")

# 可视化（使用 Open3D）
pcd = o3d.geometry.PointCloud()
pcd.points = o3d.utility.Vector3dVector(xyz)
# 用 intensity 着色
colors = np.zeros((len(xyz), 3))
colors[:, 0] = intensity / intensity.max()  # 红色通道表示强度
pcd.colors = o3d.utility.Vector3dVector(colors)
o3d.visualization.draw_geometries([pcd])
```

### 读取标注文件（label_2/*.txt）

```python
def load_kitti_label(label_path):
    """加载 KITTI 3D 目标检测标注"""
    objects = []
    with open(label_path) as f:
        for line in f:
            parts = line.strip().split()
            obj = {
                'type':       parts[0],           # Car/Pedestrian/Cyclist
                'truncated':  float(parts[1]),     # 0-1，截断程度
                'occluded':   int(parts[2]),       # 0-3，遮挡程度
                'alpha':      float(parts[3]),     # 观测角
                'bbox_2d':    list(map(float, parts[4:8])),   # 2D 框 [x1,y1,x2,y2]
                'dimensions': list(map(float, parts[8:11])),  # [h,w,l] 米
                'location':   list(map(float, parts[11:14])), # [x,y,z] 相机坐标系
                'rotation_y': float(parts[14]),    # y 轴旋转角
            }
            objects.append(obj)
    return objects

objs = load_kitti_label("label_2/000000.txt")
for obj in objs:
    if obj['type'] != 'DontCare':
        print(f"{obj['type']}: 位置={obj['location']}, 尺寸={obj['dimensions']}")
```

## ShapeNet 数据加载代码

```python
import torch
from torch.utils.data import Dataset
import numpy as np
import os, json

class ShapeNetPartDataset(Dataset):
    """ShapeNet Part 数据集加载（用于点云分类/部件分割）"""
    def __init__(self, root, split='train', n_points=2048):
        self.root = root
        self.n_points = n_points
        self.split = split

        # 读取文件列表
        split_file = os.path.join(root, f'train_test_split/shuffled_{split}_file_list.json')
        with open(split_file) as f:
            self.file_list = json.load(f)

        # 类别映射
        self.cat2id = {}
        with open(os.path.join(root, 'synsetoffset2category.txt')) as f:
            for line in f:
                parts = line.strip().split()
                self.cat2id[parts[0]] = parts[1]

    def __len__(self):
        return len(self.file_list)

    def __getitem__(self, idx):
        file_path = os.path.join(self.root, self.file_list[idx])
        data = np.loadtxt(file_path + '.pts').astype(np.float32)

        # 随机采样固定数量点
        if len(data) > self.n_points:
            idx_sample = np.random.choice(len(data), self.n_points, replace=False)
        else:
            idx_sample = np.random.choice(len(data), self.n_points, replace=True)
        points = data[idx_sample, :3]   # xyz

        # 归一化到单位球
        centroid = points.mean(axis=0)
        points -= centroid
        scale = np.max(np.linalg.norm(points, axis=1))
        points /= scale

        return torch.from_numpy(points), idx  # (N, 3), label

dataset = ShapeNetPartDataset(root="/data/shapenet", split='train')
print(f"数据集大小: {len(dataset)}")
points, label = dataset[0]
print(f"点云形状: {points.shape}")  # [2048, 3]
```

## 深度估计评估代码

深度估计的标准评估指标：AbsRel、RMSE、δ（阈值精度）

```python
import numpy as np

def compute_depth_metrics(pred, gt, min_depth=0.1, max_depth=80.0):
    """
    计算深度估计标准评估指标
    pred, gt: numpy array [H, W]，单位为米
    """
    # 有效像素掩码
    valid = (gt > min_depth) & (gt < max_depth)
    pred = pred[valid]
    gt   = gt[valid]

    # 尺度不变评估（预测深度可能有全局尺度偏差）
    scale = np.median(gt) / np.median(pred)
    pred = pred * scale

    thresh = np.maximum(gt / pred, pred / gt)

    # 阈值精度（δ）
    delta1 = (thresh < 1.25).mean()
    delta2 = (thresh < 1.25 ** 2).mean()
    delta3 = (thresh < 1.25 ** 3).mean()

    # 误差指标
    abs_rel = np.mean(np.abs(gt - pred) / gt)          # 绝对相对误差
    sq_rel  = np.mean(((gt - pred) ** 2) / gt)         # 平方相对误差
    rmse    = np.sqrt(np.mean((gt - pred) ** 2))       # 均方根误差
    rmse_log = np.sqrt(np.mean((np.log(gt) - np.log(pred)) ** 2))  # log RMSE

    metrics = {
        'AbsRel': abs_rel, 'SqRel': sq_rel,
        'RMSE': rmse, 'RMSE_log': rmse_log,
        'δ<1.25': delta1, 'δ<1.25²': delta2, 'δ<1.25³': delta3
    }
    return metrics

# 使用示例
pred_depth = np.random.rand(480, 640) * 10   # 模拟预测深度
gt_depth   = np.random.rand(480, 640) * 10   # 真值深度
metrics = compute_depth_metrics(pred_depth, gt_depth)
for k, v in metrics.items():
    print(f"  {k}: {v:.4f}")
```

## 用 iPhone/iPad LiDAR 采集自己的 3D 数据

iPhone 12 Pro 及更新机型、iPad Pro 均内置 LiDAR 传感器，可以采集真实的 3D 点云数据。

### 推荐采集 App

| App | 功能 | 输出格式 | 价格 |
|-----|------|----------|------|
| **Polycam** | 室内室外 3D 扫描 | OBJ / GLTF / LAS 点云 | 订阅制 |
| **Scaniverse** | 高质量 3D 重建 | OBJ / PLY / USDZ | 免费 |
| **3D Scanner App** | 快速扫描 | OBJ / STL / LAS | 免费+内购 |
| **LIDAR Scanner 3D** | 工业测量 | E57 / LAS / OBJ | 内购 |

### 采集建议

```
扫描技巧：
1. 慢速移动，保持 0.5-1.5m 距离
2. 多角度覆盖（水平+俯仰各方向）
3. 光线充足（LiDAR 在暗处仍工作，但 RGB 对齐效果更好）
4. 避免全反射面（镜子/高光金属）
5. 每次扫描建议不超过 3 分钟，分批进行
```

### 将扫描结果加载到 Python

```python
import open3d as o3d
import numpy as np

# 加载 PLY 格式点云（Scaniverse 导出）
pcd = o3d.io.read_point_cloud("scan.ply")
print(f"点数: {len(pcd.points)}")
print(f"是否有颜色: {pcd.has_colors()}")

# 降采样（LiDAR 点云通常很密集）
pcd_down = pcd.voxel_down_sample(voxel_size=0.01)  # 1cm 体素

# 去噪（统计离群点滤波）
pcd_clean, ind = pcd_down.remove_statistical_outlier(nb_neighbors=20, std_ratio=2.0)

# 保存为 numpy
points = np.asarray(pcd_clean.points)
colors = np.asarray(pcd_clean.colors)
print(f"清理后点数: {len(points)}")

# 可视化
o3d.visualization.draw_geometries([pcd_clean])
```

💡 **Tips**：iPhone LiDAR 的有效测距约 5m，室外大场景精度较低。适合采集小型工件（<1m）或室内房间级别的点云数据用于算法验证。精度要求高时，还是需要工业级 LiDAR（Velodyne/Ouster 等）。
